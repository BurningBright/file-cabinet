(?<=>)(?=<(?!\/))

--找到前是linux换行，后不是'320500'的光标
\n(?!320500)

--找到后不是'create'的windows换行。即\r\n(?!create)
(?!(\r\ncreate))\r\n

--平衡组匹配？？？
(?'Open'\()[2-9][0-9]{2}(?'-Open'\))\s?[0-9]{3}-[0-9]{4}(?(Open)(?!))|[2-9][0-9]{2}(-|\.)[0-9]{3}\1[0-9]{4}
<div[^>]*>[^<>]*(((?'Open'<div[^>]*>)[^<>]*)+((?'-Open'</div>)[^<>]*)+)*(?(Open)(?!))</div>

--找非词组空格
(?<![a-z,A-Z]) | (?![a-z,A-Z])
